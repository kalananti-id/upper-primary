# Program untuk melacak dan mengaudit data pribadi
import datetime
import hashlib

class PersonalDataTracker:
    def __init__(self):
        self.data_log = []
        self.privacy_score = 100
        
    def add_data_sharing(self, app_name, data_type, permission_level):
        """Catat setiap kali data dibagikan"""
        entry = {
            'timestamp': datetime.datetime.now(),
            'app': app_name,
            'data_type': data_type,
            'permission': permission_level,
            'id': self.generate_id()
        }
        self.data_log.append(entry)
        self.update_privacy_score(permission_level)
        
    def generate_id(self):
        """Generate ID unik untuk setiap entry"""
        timestamp = str(datetime.datetime.now())
        return hashlib.md5(timestamp.encode()).hexdigest()[:8]
        
    def update_privacy_score(self, permission):
        """Update skor privasi berdasarkan tingkat izin"""
        risk_scores = {
            'minimal': -2,
            'medium': -5,
            'high': -10,
            'critical': -20
        }
        self.privacy_score += risk_scores.get(permission, 0)
        
    def get_privacy_report(self):
        """Dapatkan laporan privasi"""
        if self.privacy_score >= 80:
            status = "üü¢ Sangat Aman"
        elif self.privacy_score >= 60:
            status = "üü° Cukup Aman"
        else:
            status = "üî¥ Perlu Perhatian"
            
        return f"Privacy Score: {self.privacy_score}/100 - {status}"
        
    def audit_recent_sharing(self, days=7):
        """Audit berbagi data dalam periode tertentu"""
        cutoff = datetime.datetime.now() - datetime.timedelta(days=days)
        recent = [entry for entry in self.data_log if entry['timestamp'] > cutoff]
        
        print(f"üìä Data sharing dalam {days} hari terakhir:")
        for entry in recent:
            print(f"- {entry['app']}: {entry['data_type']} ({entry['permission']})")

# Contoh penggunaan
tracker = PersonalDataTracker()
tracker.add_data_sharing("Instagram", "Location", "medium")
tracker.add_data_sharing("TikTok", "Contacts", "high")
tracker.add_data_sharing("WhatsApp", "Phone", "minimal")

print(tracker.get_privacy_report())
tracker.audit_recent_sharing()
